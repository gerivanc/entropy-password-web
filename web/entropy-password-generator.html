# Example usage of Block III - Custom Mode
# Command: python3 entropy_password_generator.py --length 15 --with-ambiguous

import secrets
import string
import math

def generate_password_custom(
    length=15,                    # Custom length (15-128)
    use_uppercase=True,           # Include uppercase
    use_lowercase=True,           # Include lowercase  
    use_digits=True,              # Include numbers
    use_special=True,             # Include special characters
    avoid_ambiguous=False         # Include ambiguous characters
):
    # Length validation (15-128 characters)
    if length < 15 or length > 128:
        raise ValueError("Length must be between 15 and 128 characters")
    
    # Basic character sets
    uppercase = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
    lowercase = 'abcdefghijklmnopqrstuvwxyz'
    digits = '0123456789'
    special = '!@#$%^&*()_+-=[]{}|;:,.<>?~`\\'
    
    # Remove ambiguous characters if requested
    if avoid_ambiguous:
        uppercase = ''.join(c for c in uppercase if c not in 'ILO')
        lowercase = ''.join(c for c in lowercase if c not in 'ilo')
        digits = ''.join(c for c in digits if c not in '01')
        special = ''.join(c for c in special if c not in '|`')
    
    # Combine characters based on selections
    all_chars = ''
    if use_uppercase: all_chars += uppercase
    if use_lowercase: all_chars += lowercase  
    if use_digits: all_chars += digits
    if use_special: all_chars += special
    
    if not all_chars:
        raise ValueError("Select at least one character type")
    
    # Generate secure password
    password = []
    
    # Ensure at least one of each selected type
    if use_uppercase:
        password.append(secrets.choice(uppercase))
    if use_lowercase:
        password.append(secrets.choice(lowercase))
    if use_digits:
        password.append(secrets.choice(digits))
    if use_special:
        password.append(secrets.choice(special))
    
    # Complete the rest
    while len(password) < length:
        password.append(secrets.choice(all_chars))
    
    # Shuffle
    secrets.SystemRandom().shuffle(password)
    
    # Calculate entropy
    entropy = math.log2(len(all_chars)) * length
    
    return ''.join(password), entropy

# Example usage
password, entropy = generate_password_custom(
    length=15,
    use_uppercase=True,
    use_lowercase=True, 
    use_digits=True,
    use_special=True,
    avoid_ambiguous=False  # --with-ambiguous
)

print(f"Password: {password}")
print(f"Entropy: {entropy:.2f} bits")
