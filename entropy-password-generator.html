import secrets
import string
import math

def generate_password_custom(
    length=15,                    # Comprimento personalizado (15-128)
    use_uppercase=True,           # Incluir mai√∫sculas
    use_lowercase=True,           # Incluir min√∫sculas  
    use_digits=True,              # Incluir n√∫meros
    use_special=True,             # Incluir caracteres especiais
    avoid_ambiguous=False         # Incluir caracteres amb√≠guos
):
    # Valida√ß√£o do comprimento (15-128 caracteres)
    if length < 15 or length > 128:
        raise ValueError("O comprimento deve estar entre 15 e 128 caracteres")
    
    # Conjuntos de caracteres b√°sicos
    uppercase = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
    lowercase = 'abcdefghijklmnopqrstuvwxyz'
    digits = '0123456789'
    special = '!@#$%^&*()_+-=[]{}|;:,.<>?~`\\'
    
    # Remover caracteres amb√≠guos se solicitado
    if avoid_ambiguous:
        uppercase = ''.join(c for c in uppercase if c not in 'ILO')
        lowercase = ''.join(c for c in lowercase if c not in 'ilo')
        digits = ''.join(c for c in digits if c not in '01')
        special = ''.join(c for c in special if c not in '|`')
    
    # Combinar caracteres baseado nas sele√ß√µes
    all_chars = ''
    if use_uppercase: all_chars += uppercase
    if use_lowercase: all_chars += lowercase  
    if use_digits: all_chars += digits
    if use_special: all_chars += special
    
    if not all_chars:
        raise ValueError("Selecione pelo menos um tipo de caractere")
    
    # Gerar senha segura
    password = []
    
    # Garantir pelo menos um de cada tipo selecionado
    if use_uppercase:
        password.append(secrets.choice(uppercase))
    if use_lowercase:
        password.append(secrets.choice(lowercase))
    if use_digits:
        password.append(secrets.choice(digits))
    if use_special:
        password.append(secrets.choice(special))
    
    # Completar o restante
    while len(password) < length:
        password.append(secrets.choice(all_chars))
    
    # Embaralhar
    secrets.SystemRandom().shuffle(password)
    
    # Calcular entropia
    entropy = math.log2(len(all_chars)) * length
    
    return ''.join(password), entropy

# Exemplo de uso
senha, entropia = generate_password_custom(

    use_lowercase=True,

    use_special=True,
